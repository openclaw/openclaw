    ```typescript
    async nextTaskId(projectDir: string): Promise<string> {
      const tasksDir = path.join(projectDir, "tasks");
      let entries: string[];
      try {
        entries = await fs.readdir(tasksDir);
      } catch {
        // tasks/ doesn't exist — start at 1
        return "TASK-001";
      }

      const pattern = /^TASK-(\d+)\.md$/;
      let maxId = 0;
      for (const entry of entries) {
        const match = pattern.exec(entry);
        if (match) {
          const num = parseInt(match[1], 10);
          if (num > maxId) maxId = num;
        }
      }

      const next = maxId + 1;
      return `TASK-${String(next).padStart(3, "0")}`;
    }
    ```

    Key details (per D-05, D-06):
    - Scoped per project directory — each project/sub-project has independent sequence
    - Always find max existing and increment — never reuse lower IDs
    - Regex matches `TASK-NNN.md` filenames only — ignores .gitkeep, README.md, any non-task files
    - Pad to at least 3 digits with `padStart(3, "0")` — but naturally grows beyond 3 (TASK-1000) since padStart only pads when shorter
    - Returns the ID string only (e.g., "TASK-001"), not the filename
