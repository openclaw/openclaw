<feature>
  <name>Domain Blocklist Module (domain-filter.ts)</name>
  <files>src/infra/net/domain-filter.ts, src/infra/net/domain-filter.test.ts</files>
  <behavior>
    ## Exact matching (MATCH-01)
    - isDomainBlocked("malware.test") -> true (in default list)
    - isDomainBlocked("phishing.test") -> true (in default list)
    - isDomainBlocked("blocked.bad") -> true (in default list)

    ## Subdomain matching (MATCH-02)
    - isDomainBlocked("sub.malware.test") -> true (malware.test in list)
    - isDomainBlocked("deep.sub.phishing.test") -> true (phishing.test in list)
    - isDomainBlocked("a.b.c.tracker.test") -> true (tracker.test in list)

    ## Non-blocked domains (no false positives)
    - isDomainBlocked("example.com") -> false
    - isDomainBlocked("google.com") -> false
    - isDomainBlocked("notmalware.test") -> false (different domain, same TLD)
    - isDomainBlocked("safe.test.example.com") -> false ("test" appears but not as suffix match)

    ## Normalization (MATCH-03)
    - isDomainBlocked("MALWARE.TEST") -> true (case insensitive)
    - isDomainBlocked("malware.test.") -> true (trailing dot)
    - isDomainBlocked("  malware.test  ") -> true (whitespace trimming)
    - isDomainBlocked("") -> false (empty string)
    - isDomainBlocked("   ") -> false (whitespace only)

    ## Default list (LIST-01)
    - DEFAULT_BLOCKED_DOMAINS contains 7 test-safe domains using .test and .bad TLDs
    - DEFAULT_BLOCKED_DOMAINS is readonly string[]

    ## Atomic Set (LIST-02)
    - setBlockedDomains(["new.test"]) replaces entire blocklist atomically
    - After setBlockedDomains(["new.test"]), isDomainBlocked("malware.test") -> false
    - After setBlockedDomains(["new.test"]), isDomainBlocked("new.test") -> true
    - addBlockedDomain("extra.test") adds a single domain
    - removeBlockedDomain("malware.test") removes a single domain
    - Mutators normalize input (addBlockedDomain("EXTRA.TEST") then isDomainBlocked("extra.test") -> true)
    - Empty/whitespace domains are filtered out by mutators

    ## Error class (OBS-01)
    - new DnsBlocklistError("malware.test") has message "Domain blocked by DNS blocklist: malware.test"
    - DnsBlocklistError instanceof SsrFBlockedError -> true
    - DnsBlocklistError.name === "DnsBlocklistError"
  </behavior>
  <implementation>

    6. **addBlockedDomain(domain: string): void** -- normalize, guard on empty, add to Set.

    7. **removeBlockedDomain(domain: string): void** -- normalize, guard on empty, delete from Set.

    All imports use .js extensions (ESM). Import normalizeHostname from "./hostname.js" and SsrFBlockedError from "./ssrf.js".
