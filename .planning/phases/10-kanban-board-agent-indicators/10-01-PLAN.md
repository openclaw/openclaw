async getTaskCheckpoint(projectName: string, taskId: string): Promise<CheckpointData | null> {
  const projectDir = this.projectService.getProjectDir(projectName);
  if (!projectDir) return null;
  const taskFile = path.join(projectDir, "tasks", `${taskId}.md`);
  const cpPath = checkpointPath(taskFile);

}

```typescript
"projects.task.checkpoint.get": async ({ params, respond }) => {
  const projectName = params?.project as string;
  const taskId = params?.taskId as string;
  if (!projectName || !taskId) {
    respond({ error: "Missing project or taskId parameter" });
    return;
  }
  try {
    const checkpoint = await projectGateway.getTaskCheckpoint(projectName, taskId);
    respond({ checkpoint });
  } catch (err) {
    respond({ error: String(err) });
  }
},
```

Add `loadTaskCheckpoint` function after `loadProjectDashboard`:
```typescript
export async function loadTaskCheckpoint(
  state: Pick<ProjectsControllerState, "client" | "projectsCheckpoint" | "projectsCheckpointLoading"> & {
    projectsCheckpoint: Record<string, unknown> | null;
    projectsCheckpointLoading: boolean;
  },
  projectName: string,
  taskId: string,
): Promise<void> {
  if (!state.client) return;
  state.projectsCheckpointLoading = true;
  try {
    const result = await state.client.request<{ checkpoint: CheckpointData | null }>(
      "projects.task.checkpoint.get",
      { project: projectName, taskId },
    );
    state.projectsCheckpoint = result.checkpoint as Record<string, unknown> | null;
  } catch (err) {
    state.projectsCheckpoint = null;
  } finally {
    state.projectsCheckpointLoading = false;
  }
}
```

onSwitchSubView: (subView: "overview" | "board") => {
  state.projectsSubView = subView;
  state.projectsBoardExpanded = null;
  state.projectsCheckpoint = null;
  const url = new URL(window.location.href);
  const basePath = state.projectsSubProject
    ? `${state.basePath}/projects/${state.projectsName}/sub/${state.projectsSubProject}`
    : `${state.basePath}/projects/${state.projectsName}`;
  url.pathname = subView === "board" ? `${basePath}/board` : basePath;
  window.history.pushState({}, "", url.toString());
},
onTogglePeek: (taskId: string) => {
  if (state.projectsBoardExpanded === taskId) {
    state.projectsBoardExpanded = null;
    state.projectsCheckpoint = null;
  } else {
    state.projectsBoardExpanded = taskId;
    void import("./controllers/projects.ts").then((mod) =>
      mod.loadTaskCheckpoint(
        state as unknown as Parameters<typeof mod.loadTaskCheckpoint>[0],
        state.projectsName!,
        taskId,
      ),
    );
  }
},
